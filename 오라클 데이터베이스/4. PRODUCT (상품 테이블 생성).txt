   "PRO_NO" NUMBER   NOT NULL,
   "PRO_NAME" VARCHAR2(100) NOT NULL,
   "PRO_CATE_CODE" CHAR(1) NOT NULL,
   "PRO_MIDD_NO" NUMBER NOT NULL,
